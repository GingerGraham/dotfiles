# zshrc-migration-strip-patterns.txt
#
# Patterns stripped silently during .zshrc migration because dotfiles handles
# them natively. One pattern per line. Lines starting with # are comments.
# Blank lines are ignored.
#
# Matching is substring (grep -F), not regex. Keep patterns specific enough
# to avoid false positives — when in doubt, let content fall through to the
# commented migration block rather than silently dropping it.
#
# To add a new pattern after encountering a distro:
#   1. Add the line(s) below with an explanatory comment
#   2. Commit — no task logic changes needed
#
# ── Manjaro / Arch ────────────────────────────────────────────────────────────
# These variables are internal to the Manjaro zsh config and meaningless once
# manjaro-zsh-config is no longer sourced directly by the user's .zshrc.
# dotfiles sources manjaro-zsh-config via distro/arch.sh instead.
USE_POWERLINE=
HAS_WIDECHARS=
# The manjaro-zsh-config fragment is sourced by distro/arch.sh
source /usr/share/zsh/manjaro-zsh-config
# The manjaro prompt is handled by the prompt engine election in loader.sh
source /usr/share/zsh/manjaro-zsh-prompt
